#! /bin/bash

echo '////////////////////////////////////////////////////////////////////////'
ps -u $(whoami) | wc -l | sed -re 's/^/Запущенных\ процессов:\ /'

echo '////////////////////////////////////////////////////////////////////////'
df -h /
echo '////////////////////////////////////////////////////////////////////////'
